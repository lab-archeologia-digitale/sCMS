import MyDataTb from "../modules/datatb.js"

`<MyDataTb />` component can be used in MDX pages to display a dynamic data table, which are fully customisable.

## Parameters

- `path2cvs` (**required** if `dTable` id not provided): relative or full path to a csv file to use for the data Table creation
- `dEndPoint` (**required** is `path2csv` or `dTable` are not defined): full path to a Directus endpoint, complete with table name
- `dToken` (**required** if `dEndPoint` is provided): Directus access token, required if table is not public. If not proved GATSBY_DIRECTUS_TOKEN environment variable will be used
- `dTable` (**required**, if `path2csv` or `dEndPoint` are not provided): name of the Directus table containing geographical data
- `dColumns (**required**): list of database columns and fields`
- `dLimit  (not required, if the number are not provided is 100 by default): limit number of records extracted from API`

{<MyDataTb
dEndPoint="https://landscapearchaeology.eu/db/items/toponimi?limit=1000&offset=0"
dToken="sxE2XHdJMWJiymA9w2QJFGJWG9AAnq6E"
dLimit="1000"
dColumns={[
{
name: "ID luogo",
selector: (row, i) => row.id,
sortable: true,
},
{
name: "Toponimo",
selector: (row, i) => row.toponimi,
sortable: true,
cell: row => (

<a href={`${process.env.GATSBY_BASE_PATH}/articles?toponimo=${encodeURIComponent(row.toponimo)}`}>
{row.toponimo}
</a>
),
},
{
name: "Provincia",
selector: (row, i) => row.provincia,
sortable: true,
},
{
name: "Comune",
selector: (row, i) => row.comune,
sortable: true,
},
]}
/>}
