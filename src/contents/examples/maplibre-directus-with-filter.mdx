### Maplibre: Adding a vector layer from a Directus API with filter

Only Hellenistic and Roman sites are being displayed

<div className="row mt-3 mb-5 border-top border-bottom py-3">
  <div className="col-6">
    <MapCompLibre 
        center="37.00416,37.48551,9"
        interactiveLayerIds={["ksa-directus"]}>

      <SourceLayer
        id="2"
        type="geojson"
        dEndPoint="https://db.lad-sapienza.it/curry/items/scms_ksa"
        dFilter="filter[Hellenistic_Roman][_eq]=true"
        dGeoField="geometry"
        layerstyle={{
          id: "ksa-directus",
          type: "circle",
          paint: {
            "circle-radius": 10,
            "circle-color": "red",
          },
        }}
      />

    </MapCompLibre>
